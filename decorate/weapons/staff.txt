actor BPStaff : BPWeapon replaces Chainsaw
{
  Weapon.Kickback 0
  Weapon.SelectionOrder 2200
  Weapon.Slotnumber 1
  Weapon.UpSound ""
  Weapon.ReadySound ""
  Inventory.PickupMessage "$GOT_STAFF"
  Obituary "$OB_STAFF"
  Tag "$TAG_STAFF"
  +WEAPON.MELEEWEAPON
  States
  {
  Ready:
    BSTF A 4 A_WeaponReady
    Loop
  Deselect:
    BSTF A 1 A_Lower
    Loop
  Select:
    BSTF A 1 A_Raise
    Loop
  Fire:
	BSTF BC 2
  Hold:
    BSTF DEF 4 A_Saw("","")
    SAWG B 0 A_ReFire
	BSTF GB 2
    Goto Ready
  Spawn:
    PICK A -1
    Stop
  }
}
