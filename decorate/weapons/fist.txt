actor fisttoken : inventory
{
Inventory.Maxamount 1
+IGNORESKILL
+UNDROPPABLE

}

ACTOR BPFist : BPWeapon
{
  Weapon.SelectionOrder 3700
  Weapon.Slotnumber 1
  Weapon.Kickback 100
  Obituary "$OB_MPFIST"
  Tag "$TAG_FIST"
  +WEAPON.WIMPY_WEAPON
  +WEAPON.MELEEWEAPON
  +UNDROPPABLE
  States
  {
  Ready:
    BSOP A 1 A_WeaponReady
    Loop
  Deselect:
    BSOP A 1 A_Lower
    Loop
  Select:
    BSOP A 1 A_Raise
    Loop
  Fire:
	  Leftswing:
		TNT1 A 0 A_JumpifInventory("fisttoken",1,"Rightswing")
		BFIL A 1
		BFIL B 1 A_Punch
		BFIL C 2
		BFIL D 2
		BFIL E 2 
		BFIL EFGH 2
		TNT1 A 0 A_Giveinventory("fisttoken")
		TNT1 A 0 A_ReFire
		TNT1 A 3
		BSOP B 1
		Goto Ready
	  Rightswing:
		BFIR A 1
		BFIR B 1 A_Punch
		BFIR C 2
		BFIR D 2
		BFIR E 2 
		BFIR EFGH 2
		TNT1 A 0 A_Takeinventory("fisttoken")		
		TNT1 A 0 A_ReFire
		TNT1 A 3
		BSOP B 1
		Goto Ready
  }
}
