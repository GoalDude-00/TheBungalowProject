actor fisttoken : inventory
{
Inventory.Maxamount 1
+IGNORESKILL
+UNDROPPABLE

}

ACTOR BPFist : BPWeapon
{
  Weapon.SelectionOrder 3700
  Weapon.Slotnumber 1
  Weapon.Kickback 100
  Obituary "$OB_BFIST" //brettyfist
  Tag "$TAG_FIST"
  +WEAPON.WIMPY_WEAPON
  +WEAPON.MELEEWEAPON
  +UNDROPPABLE
  States
  {
  Ready:
    BSOP A 1 A_WeaponReady
    Loop
  Deselect:
    BSOP A 1 A_Lower
    Loop
  Select:
    BSOP A 1 A_Raise
    Loop
  Fire:
	  LeftSwing:
		TNT1 A 0 A_JumpifInventory("PowerStrength",1,"LeftSwingZerk")
		TNT1 A 0 A_JumpifInventory("fisttoken",1,"RightSwing")
		BFIL A 1 A_Playsound("skeleton/swing", CHAN_AUTO)
		BFIL B 1 A_Punch
		BFIL C 1
		BFIL D 1
		BFIL E 2 
		BFIL EFGH 2
		TNT1 A 0 A_Giveinventory("fisttoken")
		TNT1 A 0 A_ReFire
		TNT1 A 3
		BSOP B 1
		Goto Ready
	  Rightswing:
		BFIR A 1 A_Playsound("skeleton/swing", CHAN_AUTO)
		BFIR B 1 A_Punch
		BFIR C 1
		BFIR D 1
		BFIR E 2 
		BFIR EFGH 2
		TNT1 A 0 A_Takeinventory("fisttoken")		
		TNT1 A 0 A_ReFire
		TNT1 A 3
		BSOP B 1
		Goto Ready
	  //RRRAAAAAAAAAAARRRGGHHHHHH
	  //WHAT YOU LOOKIN' AT?
	  LeftSwingZerk:
		TNT1 A 0 A_JumpifInventory("fisttoken",1,"RightSwingZerk")
		BSTL A 1 A_Playsound("skeleton/swing", CHAN_AUTO)
		BSTL B 1 A_Punch
		BSTL C 2
		BSTL D 2
		BSTL E 2 
		BSTL EFGH 2
		TNT1 A 0 A_Giveinventory("fisttoken")
		TNT1 A 0 A_ReFire
		TNT1 A 3
		BSOP B 1
		Goto Ready
	  RightSwingZerk:
		BSTR A 1 A_Playsound("skeleton/swing", CHAN_AUTO)
		BSTR B 1 A_Punch
		BSTR C 2
		BSTR D 2
		BSTR E 2 
		BSTR EFGH 2
		TNT1 A 0 A_Takeinventory("fisttoken")		
		TNT1 A 0 A_ReFire
		TNT1 A 3
		BSOP B 1
		Goto Ready
  }
}
