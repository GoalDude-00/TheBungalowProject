ACTOR BPAR : BPWeapon replaces Chaingun
{
		Weapon.SelectionOrder 700
		Weapon.AmmoUse 1
		Weapon.AmmoGive 20
		Weapon.AmmoType "Clip"
		Inventory.PickupMessage "$GOTCHAINGUN"
		Obituary "$OB_MPCHAINGUN"
		Tag "$TAG_CHAINGUN"
		Weapon.Slotnumber 4
   States
   {
      Select:
		BGAR A 1 A_Raise
		Loop
	  Deselect:
		BGAR A 1 A_Lower
		Loop
	  Ready:
		BGAR A 1 A_WeaponReady(WRF_AllOWRELOAD)
		Loop
	  Fire:
		TNT1 A 0 A_Gunflash	
		TNT1 A 0 A_Jump(96,"PFire2","PFire3","PFire4")	
	PFire1:
		BFLS A 1 BRIGHT A_FireCGun
		BGAR A 1
		BGAR B 1
		BGAR C 1
		goto Postshot
	PFire2:	
		BFLS B 1 BRIGHT A_FireCGun
		BGAR A 1
		BGAR B 1
		BGAR C 1
		goto Postshot
	PFire3:	
		BFLS C 1 BRIGHT A_FireCGun
		BGAR A 1
		BGAR B 1
		BGAR C 1
		goto Postshot
	PFire4:	
		BFLS D 1 BRIGHT A_FireCGun
		BGAR A 1
		BGAR B 1
		BGAR C 1		
	Postshot:
		TNT1 A 0 A_Refire	
		BGAR C 2
		BGAR D 2
		BGAR E 2
		Goto Ready

	Flash:
		TNT1 A 5 Bright A_Light1
		Goto LightDone
		TNT1 B 5 Bright A_Light2
		Goto LightDone

		Spawn:
			PICK E -1
			stop
   }
}
