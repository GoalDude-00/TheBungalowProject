ACTOR BPHPG : BPWeapon replaces PlasmaRifle
{
	Weapon.SelectionOrder 100
	Weapon.AmmoUse 1
	Weapon.AmmoGive 40
	Weapon.AmmoType "Cell"
	Inventory.PickupMessage "$GOT_HPG"
	Tag "$TAG_HPG"
	States
	{
	Ready:
		BGPI AABBCCDDEEFFGGHH 1 A_WeaponReady
		Loop
	Deselect:
		BGPI A 1 A_Lower
		Loop
	Select:
		BGPI A 1 A_Raise
		Loop
	Fire:
		BGPF A 0 A_GunFlash
		BGPF A 0 A_FireCustomMissile("BPPlasmaBall")
		BGPF ABC 1 BRIGHT
		BGPF D 5 A_ReFire
		BGPF EF 5
		BGPI A 5
		Goto Ready
	Flash:
		TNT1 A 2 Bright A_Light1
		TNT1 A 2 Bright A_Light2
		TNT1 A 2 Bright A_Light1
		Goto LightDone
	Spawn:
		PICK G -1
		Stop
	}
}

ACTOR BPPlasmaBall : PlasmaBall
{
	RenderStyle Normal
	Alpha 1
	SeeSound "weapons/plasmaf"
	DeathSound "weapons/plasmax"
	Obituary "$OB_HPG"
	States
	{
	Spawn:
		FIRB ABC 2 Bright
		Loop
	Death:
		FIRD ABCD 4 Bright
		FIRD EF 2 Bright
		Stop
	}
}
