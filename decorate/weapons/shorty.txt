ACTOR BPSuperShotgun : BPWeapon replaces SuperShotgun
{
	Weapon.SelectionOrder 400
	Weapon.AmmoUse 2
	Weapon.AmmoGive 8
	Weapon.AmmoType "Shell"
	Inventory.PickupMessage "$GOT_SHORTY"
	Obituary "$OB_SHORTY"
	Tag "$TAG_SHORTY"
	States
	{
	Ready:
		BSSI A 1 A_WeaponReady //BuSSIn
		Loop
	Deselect:
		BSSI A 1 A_Lower
		Loop
	Select:
		BSSI A 1 A_Raise
		Loop
	Fire:
		BSSI A 3
		BSSF A 0 A_FireShotgun2
		BSSF AB 1 BRIGHT
		BSSF CDEFG 1
		BSSI A 9
		BSSI A 1 A_CheckReload
		BSSR A 1
		BSSR B 1
		BSSR C 2 A_OpenShotgun2
		BSSR D 1
		BSSR EFGH 1
		BSSR I 2
		BSSR J 1
		BSSR K 2 A_LoadShotgun2
		BSSR L 1
		BSSR MNO 2
		BSSR P 0 A_CloseShotgun2
		BSSR PQRSTUV 1
		BSSI A 5 A_ReFire
		Goto Ready
	// unused states
		SHT2 B 7
		SHT2 A 3
		Goto Deselect
	Flash:
		TNT1 I 4 Bright A_Light1
		TNT1 J 3 Bright A_Light2
		Goto LightDone
	Spawn:
		PICK D -1
		Stop
	}
}
